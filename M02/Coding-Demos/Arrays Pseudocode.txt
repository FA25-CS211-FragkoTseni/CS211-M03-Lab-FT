// Declare an array of size N (or dynamic if allowed)
DECLARE array AS ARRAY OF INTEGER

// Insert element at the end
PROCEDURE InsertAtEnd(array, value)
    APPEND value TO array
END PROCEDURE

// Insert element at the beginning
PROCEDURE InsertAtBeginning(array, value)
    SHIFT all elements one position to the right
    SET array[0] = value
END PROCEDURE

// Insert element in the middle (at position pos)
PROCEDURE InsertAtMiddle(array, pos, value)
    SHIFT all elements from pos to end one position to the right
    SET array[pos] = value
END PROCEDURE

// Delete element from the end
PROCEDURE DeleteFromEnd(array)
    REMOVE last element FROM array
END PROCEDURE

// Delete element from the beginning
PROCEDURE DeleteFromBeginning(array)
    SHIFT all elements one position to the left
    REMOVE last element
END PROCEDURE

// Delete element from the middle (at position pos)
PROCEDURE DeleteFromMiddle(array, pos)
    FOR i FROM pos TO LENGTH(array) - 2
        SET array[i] = array[i + 1]
    END FOR
    REMOVE last element
END PROCEDURE

// Search for an element
PROCEDURE SearchElement(array, value)
    FOR i FROM 0 TO LENGTH(array) - 1
        IF array[i] == value THEN
            RETURN i // index found
        END IF
    END FOR
    RETURN -1 // not found
END PROCEDURE

// Display the array
PROCEDURE DisplayArray(array)
    FOR EACH element IN array
        PRINT element
    END FOR
END PROCEDURE
