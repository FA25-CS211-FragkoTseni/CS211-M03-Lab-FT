// Declare a dictionary
DECLARE dict AS DICTIONARY

// Insert key-value pair
PROCEDURE InsertKeyValue(dict, key, value)
    SET dict[key] = value
END PROCEDURE

// Update value for an existing key
PROCEDURE UpdateValue(dict, key, newValue)
    IF key EXISTS IN dict THEN
        SET dict[key] = newValue
    ELSE
        PRINT "Key not found"
    END IF
END PROCEDURE

// Delete a key-value pair
PROCEDURE DeleteKey(dict, key)
    IF key EXISTS IN dict THEN
        REMOVE key FROM dict
    ELSE
        PRINT "Key not found"
    END IF
END PROCEDURE

// Search for a value by key
PROCEDURE SearchByKey(dict, key)
    IF key EXISTS IN dict THEN
        RETURN dict[key]
    ELSE
        RETURN "Key not found"
    END IF
END PROCEDURE

// Display the dictionary
PROCEDURE DisplayDictionary(dict)
    FOR EACH key, value IN dict
        PRINT key, ":", value
    END FOR
END PROCEDURE
